class ObjektPrinter
{
  public void ausgeben( String st )
    // 3. Der Objektreferenzparameter st
    // bezieht sich auf das eine Objekt.
  {
    System.out.println("Wert des Parameters: " + st );
  }   // 4. Die Objektreferenz in st
      // kann wie üblich verwendet werden.
}



class OPTester
{
  public static void main ( String[] args )
  {
    String mitteilung = "Nur ein Objekt" ;
      // 1. Die Objektreferenzvariable
      // mitteilung verweist auf das Objekt.

    ObjektPrinter op = new ObjektPrinter();

    System.out.println
      ("Erster  Wert von mitteilung: " + mitteilung );
    op.ausgeben( mitteilung );
      // 2. Beim Aufruf der Methode wird die
      // Objektreferenz  als Parameterwert übergeben.

    System.out.println
      ("Zweiter Wert von mitteilung: " + mitteilung );
  }
}
